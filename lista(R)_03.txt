#ifdef ex05

#include <stdio.h>
#include <string.h>

int fatorial(int n) {
    int fat = 1;
    for (int i = 1; i<=n; i++) {
        fat *= i;
    }
    return fat;
}

int sumDigits(int num) {
    int sum = 0;

    char buffer[10];
    sprintf(buffer, "%d", num);

    for (int i = 0; i<strlen(buffer); i++) {
        int charInt = buffer[i] - '0';
        sum += charInt;
    }
    return sum;
}

int main() {

    int n = 4; //Valor que será fatorado
    int fat = fatorial(n);
    int sum = sumDigits(fat);



    printf("n: %d\n", n);
    printf("fat: %d\n", fat);
    printf("sum: %d\n", sum);
}
